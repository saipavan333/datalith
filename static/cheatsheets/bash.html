Files & navigation

📂 Navigate & manage

pwd              # where am I
ls -lah          # long, all, human sizes
cd /path   cd -  # go / previous dir
mkdir -p a/b/c   # make nested
cp -r src dst    mv a b     rm -rf dir
ln -s target link # symlink
find . -name "*.csv" -mtime -1  # modified <1 day
du -sh *   df -h          # sizes / disk free

👀 View & inspect

cat f        less f        # view / page
head -n 20 f    tail -n 20 f
tail -f app.log      # follow live
wc -l f              # line count
sort f | uniq -c    # dedupe + counts
diff a b    md5sum f
column -t -s, f      # pretty CSV

🔀 Pipes & redirection

cmd1 | cmd2            # pipe stdout -> stdin
cmd > out.txt          # stdout to file (overwrite)
cmd >> out.txt         # append
cmd 2> err.txt         # stderr to file
cmd > all.txt 2>&1     # both streams
cmd1 && cmd2           # run 2 if 1 succeeds
cmd1 || cmd2           # run 2 if 1 fails
cmd < input.txt        # file to stdin
Text crunching

🔎 grep

grep "error" app.log
grep -i -n "warn" f     # ignore case, line #
grep -r "TODO" src/     # recursive
grep -v "debug" f       # invert (exclude)
grep -c "200" f         # count matches
grep -E "a|b" f          # regex (extended)
grep -oE "[0-9]+" f       # only the match

🧮 awk

# print column 3 (space-delimited)
awk '{print $3}' f
# CSV: field sep = comma
awk -F, '{print $1, $3}' f
# filter + sum a column
awk -F, '$2 > 100 {s += $2} END {print s}' f
# count by key
awk -F, '{c[$1]++} END {for (k in c) print k, c[k]}' f

✂️ sed, cut, sort, uniq

sed 's/old/new/g' f        # substitute all
sed -n '1,10p' f            # print lines 1-10
sed -i 's/a/b/g' f          # edit in place

cut -d, -f1,3 f            # CSV cols 1,3
sort -t, -k2 -n -r f       # by col2, numeric, desc
uniq -c   uniq -d         # counts / dups only
tr 'a-z' 'A-Z' < f       # translate

🧷 xargs & find combos

# delete all matching files
find . -name "*.tmp" | xargs rm -f
# safer with null-delimit (spaces)
find . -name "*.csv" -print0 | xargs -0 gzip
# parallel
cat urls.txt | xargs -P4 -n1 curl -O
Scripting

$ Variables & expansion

name="ada"          # no spaces around =
echo "$name"          # quote to be safe
files=$(ls *.csv)      # command substitution
count=$((2 + 3))       # arithmetic
echo ${VAR:-default}   # default if unset
export DB_URL="…"     # env var for children
"$1" "$@" $#           # args / all / count

🔁 Conditionals & loops

if [ -f "$f" ]; then echo "exists"; fi
# -f file, -d dir, -z empty, -n non-empty
[ "$a" = "$b" ]   [ $n -gt 5 ]  # -eq -ne -lt -ge

for f in *.csv; do echo "$f"; done
while read line; do echo "$line"; done < f
case "$x" in a) …;; *) …;; esac

📜 Robust scripts

#!/usr/bin/env bash
set -euo pipefail   # exit on error/unset/pipe-fail
IFS=$'\n\t'

log() { echo "[$(date +%F_%T)] $*"; }
trap 'log "failed at line $LINENO"' ERR

main() { log "start"; }
main "$@"

Exit code 0 = success; check $?. set -euo pipefail is the safety standard.

🔐 Permissions & processes

chmod +x run.sh       chmod 644 f   # rw-r--r--
chown user:grp f
ps aux | grep python
top   htop            # live processes
kill -9 PID    pkill -f name
nohup cmd &            # run detached
cmd &   jobs   fg %1    # background / foreground

🌐 Network & transfer

curl -sSL "$url" -o out.json
curl -X POST -H "Content-Type: application/json" \
     -d '{"k":1}' "$url"
wget "$url"
ssh user@host       scp f user@host:/path
rsync -avz src/ host:/dst/

🗜️ Archives & cron

tar -czf out.tar.gz dir/   # compress
tar -xzf out.tar.gz        # extract
gzip f    gunzip f.gz    zcat f.gz

# cron: min hr dom mon dow  cmd
crontab -e
0 6 * * *  /path/run.sh   # daily 06:00
*/15 * * * * cmd          # every 15 min
