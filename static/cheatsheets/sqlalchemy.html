Connect

🔌 Engine & connection

from sqlalchemy import create_engine, text

engine = create_engine(
    "postgresql+psycopg://user:pw@host:5432/db",
    pool_size=5, max_overflow=10, pool_pre_ping=True)

with engine.connect() as conn:      # auto-close
    conn.execute(text("SELECT 1"))
with engine.begin() as conn:        # transaction (commit/rollback)
    conn.execute(text("INSERT ..."))

URLs: postgresql+psycopg, mysql+pymysql, sqlite:///file.db, snowflake://....

📝 Raw SQL (parametrized)

with engine.connect() as conn:
    rows = conn.execute(
        text("SELECT * FROM t WHERE dept = :d"),
        {"d": "eng"}).fetchall()
    for r in rows:
        r.id, r.name        # Row objects, attr access
        r._mapping           # -> dict
# executemany: pass a list of dicts
conn.execute(text("INSERT INTO t (a) VALUES (:a)"),
             [{"a":1}, {"a":2}])

Always bind with :param — never f-string values into SQL.

Core (expression API)

🏗️ Define tables (metadata)

from sqlalchemy import (MetaData, Table, Column,
    Integer, String, Numeric, DateTime, ForeignKey)

md = MetaData()
users = Table("users", md,
    Column("id", Integer, primary_key=True),
    Column("name", String(100), nullable=False),
    Column("dept_id", ForeignKey("dept.id")))
md.create_all(engine)      # CREATE TABLE if not exists

🔎 Core select / insert

from sqlalchemy import select, insert, update, delete, func

stmt = (select(users.c.name, func.count().label("n"))
        .where(users.c.dept_id == 1)
        .group_by(users.c.name)
        .order_by(users.c.name))
conn.execute(stmt).all()

conn.execute(insert(users), [{"name":"Ada"}])
conn.execute(update(users).where(users.c.id==1)
             .values(name="Bo"))
conn.execute(delete(users).where(users.c.id==1))
ORM

🧱 Declarative models

from sqlalchemy.orm import (DeclarativeBase,
    Mapped, mapped_column, relationship)

class Base(DeclarativeBase): pass

class User(Base):
    __tablename__ = "users"
    id:   Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str]
    dept_id: Mapped[int] = mapped_column(ForeignKey("dept.id"))
    dept: Mapped["Dept"] = relationship(back_populates="users")

Base.metadata.create_all(engine)

💾 Session & CRUD

from sqlalchemy.orm import Session

with Session(engine) as s:
    u = User(name="Ada", dept_id=1)
    s.add(u)                 # INSERT (on flush)
    s.commit()

    u = s.get(User, 1)       # by PK
    u.name = "Bo"            # UPDATE on commit
    s.delete(u)              # DELETE
    s.commit()

🔍 ORM queries

stmt = (select(User)

users = s.scalars(stmt).all()      # list[User]

🐼 pandas & tips

import pandas as pd
df = pd.read_sql(select(User), engine)
df = pd.read_sql("SELECT * FROM users", engine)
df.to_sql("users", engine, if_exists="append",
          index=False, chunksize=10_000, method="multi")
  • Use engine.begin() for write transactions (auto commit/rollback).
  • pool_pre_ping=True avoids stale-connection errors.
  • One Session per unit of work; don't share across threads.
